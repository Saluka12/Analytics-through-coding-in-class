genre_counts = movies["genre"].value_counts()
selected_genres = genre_counts[genre_counts >= 30].index  # filter for genres with enough data

sub_all = movies[movies["genre"].isin(selected_genres)].copy()
n_genres = len(selected_genres)
rows = int(np.ceil(n_genres / 3))
cols = 3

fig, axes = plt.subplots(rows, cols, figsize=(16, 4 * rows), sharex=True, sharey=True)
axes = axes.ravel()

for i, g in enumerate(selected_genres):
    tmp = sub_all[sub_all["genre"] == g]
    axes[i].scatter(
        tmp["rating"], tmp["gross"],
        s=12, alpha=0.7,
        color=sns.color_palette("tab10")[i % 10]
    )
    axes[i].set_title(g, fontsize=12)
    axes[i].grid(True, linestyle='--', alpha=0.3)

for j in range(i + 1, len(axes)):
    axes[j].set_visible(False)

fig.text(0.5, 0.04, "Rating", ha="center", fontsize=12)
fig.text(0.04, 0.5, "Gross (USD)", va="center", rotation="vertical", fontsize=12)
fig.suptitle("Gross vs Rating by Genre (All Genres with 20 Movies)", fontsize=16, y=1.03)
plt.tight_layout(rect=[0.05, 0.05, 1, 0.95])
plt.show()
# rating does not seem to have a strong correlation with gross revenue across any genre. however, a score of 7-10 seems to be a threshold for higher gross revenue in most genres.
